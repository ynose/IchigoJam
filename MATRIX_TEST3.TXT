1 'MATRIX LED
10 CLS:CLV
20 POKE #700,#21,#81,#E1,#00
30 FOR M=#700 TO #702
40  R=I2CW(#70,M,1,#700,0)
50 NEXT

'Main Loop
60  FOR C=#0 TO #FF
70   LC 12,5:?HEX$(C,2);
80   WAIT 60

'In
100  FOR S=8 TO 1 STEP -1
110   GOSUB 310
120   FOR I=0 TO 7
130    [I]=[I]>>S
140   NEXT
150   R=I2CW(#70,#703,1,#800,8)
160   WAIT 5
170  NEXT

'Out
200  GOSUB 310
210  FOR S=1 TO 8
220   FOR I=0 TO 7
230    [I]=[I]<<1
240   NEXT
250   R=I2CW(#70,#703,1,#800,8)
260   WAIT 5
270  NEXT

280  A=A+8:B=A
290 NEXT
300 END

'Chr
400 FOR D=0 TO 7
410  [D]=PEEK(A+D)<<8+PEEK(B+D)
420  LC 12,8+D:?BIN$([D],8)
430 NEXT
440 RETURN
